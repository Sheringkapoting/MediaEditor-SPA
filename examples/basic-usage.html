3. Batch Processing

Process multiple images with the same watermark:

// Initialize batch processor const batchProcessor = new BatchProcessor(); // Set progress callback batchProcessor.setProgressCallback((progress) => { console.log(`Processing: ${progress.percent}%`); updateProgressBar(progress.percent); }); // Add images to queue const imageFiles = document.getElementById('imageInput').files; const sourceImages = []; for (const file of imageFiles) { const imageData = await imageProcessor.loadImage(file); sourceImages.push({ id: Date.now() + Math.random(), name: file.name, size: file.size, imageData: imageData }); } // Process all images batchProcessor.addToQueue(sourceImages, watermarkSettings); const results = await batchProcessor.processQueue(); console.log(`Processed ${results.completed} images successfully`);

7. Responsive Integration

Integrate with modern web frameworks:

// React Component Example function WatermarkComponent() { const [processing, setProcessing] = useState(false); const [result, setResult] = useState(null); const handleWatermark = async (file, settings) => { setProcessing(true); try { const imageProcessor = new ImageProcessor(); const watermarkEngine = new WatermarkEngine(); const imageData = await imageProcessor.loadImage(file); const watermarked = await watermarkEngine.generateWatermarkedImage(imageData, settings); setResult(watermarked.dataUrl); } catch (error) { console.error('Error:', error); } finally { setProcessing(false); } }; return ( <div> {processing && <p>Processing...</p>} {result && <img src={result} alt="Watermarked" />} </div> ); }
